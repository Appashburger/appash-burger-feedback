// script.js
document.addEventListener('DOMContentLoaded', () => {
  const form = document.getElementById('feedbackForm');
  const thanks = document.getElementById('thanks');
  const sendBtn = document.getElementById('sendBtn');
  const newFeedback = document.getElementById('newFeedback');

  form.addEventListener('submit', async (e) => {
    e.preventDefault();
    // validation basique
    const snap = form.snapchat.value.trim();
    const message = form.message.value.trim();

    if(!snap || snap.length < 2){
      alert('Merci de renseigner un nom d\'utilisateur Snapchat valide.');
      return;
    }
    if(!message || message.length < 3){
      alert('Merci de laisser un message un peu plus long.');
      return;
    }

    sendBtn.disabled = true;
    sendBtn.textContent = 'Envoi...';

    try {
      // Envoi standard HTML form POST vers Formspree (fallback)
      // Ici on utilise fetch pour rester sur la page.
      const action = form.action;
      const formData = new FormData();
      formData.append('snapchat', snap);
      formData.append('message', message);

      const res = await fetch(action, {
        method: 'POST',
        body: formData,
        headers: {
          'Accept': 'application/json'
        }
      });

      if (res.ok) {
        form.classList.add('hidden');
        thanks.classList.remove('hidden');
      } else {
        // Formspree renvoie parfois 422 si le formulaire n'est pas configuré
        const data = await res.json().catch(()=>null);
        console.error('Form error', data);
        alert('Erreur lors de l\'envoi. Vérifie ton endpoint Formspree ou consulte la console.');
      }
    } catch (err) {
      console.error(err);
      alert('Une erreur réseau est survenue lors de l\'envoi.');
    } finally {
      sendBtn.disabled = false;
      sendBtn.textContent = 'Envoyer mon retour';
    }
  });

  if(newFeedback){
    newFeedback.addEventListener('click', () => {
      // reset
      document.getElementById('snap').value = '';
      document.getElementById('message').value = '';
      thanks.classList.add('hidden');
      form.classList.remove('hidden');
    });
  }
});
